#include <stdio.h>

int main() {
    int a[] = {1, 2, 3, 4, 5};
    int evenSum = 0, oddSum = 0;

    for (int i = 0; i < 5; i++) {
        if (a[i] % 2 == 0)
            evenSum += a[i];
        else
            oddSum += a[i];
    }

    printf("Sum of even numbers: %d\n", evenSum);
    printf("Sum of odd numbers: %d\n", oddSum);

    return 0;
}
