#include <stdio.h>

int main() {
    int a[] = {2, 3, 2, 4, 3, 5, 2};
    int n = 7;
    int count[100] = {0}; // assuming elements are less than 100

    for (int i = 0; i < n; i++)
        count[a[i]]++;

    for (int i = 0; i < 100; i++) {
        if (count[i] > 0)
            printf("Element %d occurs %d times\n", i, count[i]);
    }

    return 0;
}
