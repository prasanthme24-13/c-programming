#include <stdio.h>

int main() {
    int a[] = {10, 25, 30, 15, 20};
    int n = 5;
    int first = a[0], second = -1;

    for (int i = 1; i < n; i++) {
        if (a[i] > first) {
            second = first;
            first = a[i];
        } else if (a[i] > second && a[i] != first) {
            second = a[i];
        }
    }

    printf("Second largest element: %d\n", second);
    return 0;
}
